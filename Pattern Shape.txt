#include <stdio.h>

// 1. Segitiga Sama Kaki (Isosceles)

Input: n = 5
Output:

    *
   ***
  *****
 *******
*********

void isosceles(int n){
    for(int i=0;i<n;i++){
        for(int j=0;j<n-i-1;j++) printf(" ");
        for(int j=0;j<2*i+1;j++) printf("*");
        printf("\n");
    }
}

// 2. Segitiga Sama Kaki Hollow

Input: n = 5
Output:

    *
   * *
  *   *
 *     *
*********

void isoscelesHollow(int n){
    for(int i=0;i<n;i++){
        for(int j=0;j<n-i-1;j++) printf(" ");
        for(int j=0;j<2*i+1;j++){
            if(j==0 || j==2*i || i==n-1) printf("*");
            else printf(" ");
        }
        printf("\n");
    }
}

void diamond(int n){
    for(int i=0;i<n;i++){
        for(int j=0;j<n-i-1;j++) printf(" ");
        for(int j=0;j<2*i+1;j++) printf("*");
        printf("\n");
    }
    for(int i=n-2;i>=0;i--){
        for(int j=0;j<n-i-1;j++) printf(" ");
        for(int j=0;j<2*i+1;j++) printf("*");
        printf("\n");
    }
}

// 10. Hollow Diamond

Input: n = 5
Output:

    *
   * *
  *   *
 *     *
*       *
 *     *
  *   *
   * *
    *

void hollowDiamond(int n){
    for(int i=0;i<n;i++){
        for(int j=0;j<n-i-1;j++) printf(" ");
        for(int j=0;j<2*i+1;j++){
            if(j==0 || j==2*i) printf("*");
            else printf(" ");
        }
        printf("\n");
    }
    for(int i=n-2;i>=0;i--){
        for(int j=0;j<n-i-1;j++) printf(" ");
        for(int j=0;j<2*i+1;j++){
            if(j==0 || j==2*i) printf("*");
            else printf(" ");
        }
        printf("\n");
    }
}

void rectangleHollow(int r, int c){
    for(int i=0;i<r;i++){
        for(int j=0;j<c;j++){
            if(i==0 || i==r-1 || j==0 || j==c-1) printf("*");
            else printf(" ");

}

int main(){
    int choice, n, r, c;

    printf("3. Segitiga Sama Kaki Terbalik\n");

    printf("14. Persegi Panjang\n");
    printf("15. Persegi Panjang Hollow\n");
    printf("Pilihan: ");
    scanf("%d", &choice);

    if(choice>=1 && choice<=13){
        printf("Masukkan n: ");
        scanf("%d", &n);
    } else if(choice==14 || choice==15){
        printf("Masukkan baris dan kolom: ");
        scanf("%d %d", &r, &c);
    }

    switch(choice){
        case 1: isosceles(n); break;
        case 2: isoscelesHollow(n); break;
        case 3: invertedIsosceles(n); break;
        case 4: invertedIsoscelesHollow(n); break;
        case 5: rightTriangle(n); break;

        case 13: squareHollow(n); break;
        case 14: rectangle(r,c); break;
        case 15: rectangleHollow(r,c); break;
        default: printf("Pilihan tidak valid!\n");
    }

    return 0;
}

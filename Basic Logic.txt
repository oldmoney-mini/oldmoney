10. Fibonacci

for(int i=1; i<=n; i++){
    printf("%d ", a);
    next = a+b;
    a = b;
    b = next;
}

11. Cek Bilangan Prima

int prime=1;
if(n<2) prime=0;
for(int i=2; i*i<=n; i++){
    if(n%i==0){ prime=0; break; }
}
printf(prime ? "Prima" : "Bukan Prima");

12. Palindrome (String)

int pal=1;
for(int i=0; i<strlen(s)/2; i++){
    if(s[i] != s[strlen(s)-1-i]) { pal=0; break; }
}
printf(pal ? "Palindrome" : "Bukan");

13. Sorting (Bubble Sort â†’ sering muncul)

int arr[5] = {5,2,4,1,3}, n=5;
for(int i=0;i<n-1;i++){
    for(int j=0;j<n-i-1;j++){
        if(arr[j] > arr[j+1]){
            int tmp = arr[j];
            arr[j] = arr[j+1];
            arr[j+1] = tmp;
        }
    }
}
for(int i=0;i<n;i++) printf("%d ", arr[i]);
